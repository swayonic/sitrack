364   def get_order_by(view)
365     # if there is a query id in the params, set the session value
366     if (params[:order_by])
367       order_by = params[:order_by]
368       sitrack_session.save_value('order_by', order_by)
369     end
370     # if we have a order_by in the session, use it
371     order_by = sitrack_session.get_value('order_by')
372     # Set up a default orderby
373     # if "Full Name" is a column, sort by that. Otherwise, try lastname
374     columns = view.safe_column_names
375     if columns.include?('Full_Name')
376     	default = "Full_Name asc";
377     elsif columns.include?('Last_Name')
378     	default = "Last_Name asc"
379     else
380     	# make sure there is a default
381     	default = Person.table_name+'.personID asc'
382     end
383     default += ", First_Name asc"if columns.include?('First_Name')
384     return (order_by && columns.include?(order_by.split(' ')[0])) ? order_by : default
385   end
